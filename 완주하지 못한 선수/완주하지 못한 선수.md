```java
class Solution {
    public String solution(String[] participant, String[] completion) {
        Arrays.sort(participant); //participant 배열을 오름차순으로 정렬합니다.
        Arrays.sort(completion); //completion 배열을 오름차순으로 정렬합니다.
        int i; //for문에 사용할 i를 생성합니다
        for(i = 0; i < completion.length; i++) { // 참가자 명단의 length 만큼 for문을 돌립니다.
            if(!participant[i].equals(completion[i])) { //만약 오름차순으로 정렬한 두 배열이 서로 같지 않은 idx가 있다면 break을 합니다.
                break;
            }
        }
        return participant[i]; //break 이 작용하기 전까지 실행되는 for문의 i값으로 배열에서 idx를 추출하여 return 한다
        // + for문에서 리턴을 안하는 이유는 programmers에서 제공하는 환경을 최대한 지키기 위함입니다.
        // + for문에 사용하는 i를 따로 지정한 이유는 for문 안에서 변수를 지정하면 해당 for문 안에서 밖에 사용이 안되기 때문입니다.
    }
}
```
