class Solution { 
    public String solution(int a, int b) {
        
        //요일 배열
        String[] day = {"SUN", "MON", "TUE", "WED", "THU", "FRI", "SAT"}; //요일 배열
    
        // 1일의 모든 시작이 월요일이 아니므로 ex) 1월의 시작은 일요일에서 5일이 지난 금요일이기 때문에 
        // 이 예시로 들면 1월의 idx인 plus 배열의 0번 idx를 5로 설정
        int[] plus = {5, 1, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4}; 
        
        //배열은 0부터 시작하기 때문에 month(a)값을 1로받으면 첫번째인 idx(plus[0])에 접근할 수 있도록 -1을 해주었다 
        int realPlus = plus[a-1];
        
        //plus에서 1을뺀 realPlus와 day값(b)을 더해주고 7로 나누어 준다 이유는 7일 단위로 시간이 지나면 어차피 요일은 제자리 이기 때문이다 
        //만약 1월 1일이 일요일이라 가정한다면 7일이 지나도 월요일, 14일이 지나도 월요일이기 때문이다.
        int sum = (realPlus + b) % 7;
        
        String answer = "";

        //sum(realPlus와 day(b)합)이 7의 배수로 나왔다면 그 날은 토요일이다. 하지만 나는 realPlus에서의 경우와 마찬가지로 -1을 해주었다 그 이유는
        //만약 sum이 1이 나왔다고 가정해보자 그럼 그날짜는 일요일이다. 하지만 day에 원소값으로 sum을 준다면 ? 배열에서는 0부터 시작하기때문에 "MON"을 반환할 것이다.
        //그렇기때문에 sum이 0이라면 원소값에 sum-1을 해주면 0-1이기 떄문에 ArrayIndexOutOfBoundsException 이 터지기 때문에 바로 "SAT"을 반환해주었다.
        if (sum == 0) {
            answer = "SAT";
        }
        //0이 아닐시 원래의 로직대로 -1을 함
        else answer = day[sum - 1];

        return answer;

    }
}
